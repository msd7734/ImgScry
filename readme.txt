usage: ImgScry [options] set_abbreviation
options:
 -f, --format	 Format output filenames with C-style string formatting.
 -q, --quality	 Image download quality (small, normal, large). Default is normal
 -s, --skip	 Attempt to skip files that have already been downloaded. Default is true
 -l, --limit	 Do not download more than this number of cards. Default is no limit
